A JavaScript equivalent of PHP's deg2rad
<!-- more -->

function deg2rad (angle) {
    // http://kevin.vanzonneveld.net
    // +   original by: Enrique Gonzalez
    // *     example 1: deg2rad(45);
    // *     returns 1: 0.7853981633974483
    return (angle / 180) * Math.PI;
}
